Example 1: The Button

1. The `
` Nightmare

<!-- HTML -->
<div class="custom-button" id="cta-button">
  <span class="button-label">Submit</span>
</div>

<!-- JavaScript (Manual keyboard handling, a fragile fix for a core issue) -->
<script>
  const ctaButton = document.getElementById('cta-button');

  ctaButton.addEventListener('click', () => {
    alert('Form submitted via div button!');
  });

  ctaButton.addEventListener('keydown', (event) => {
    if (event.key === 'Enter' || event.key === ' ') {
      event.preventDefault();
      ctaButton.click();
    }
  });
</script>

2. The `
` with a Life Vest

<!-- HTML with ARIA attributes -->
<div 
  class="custom-button" 
  role="button" 
  tabindex="0" 
  aria-label="Submit Form"
  onclick="alert('Form submitted via accessible div button!')"
>
  <span class="button-label">Submit</span>
</div>

3. The Best Practice (Semantic HTML)

<!-- HTML -->
<button class="native-button" type="submit" aria-label="Submit Form" onclick="alert('Form submitted via native button!')">
  Submit
</button>

Example 2: A Product Card

1. The `
` Nightmare

<!-- HTML -->
<div class="product-card">
  <div class="product-image-container">
    <img src="https://via.placeholder.com/300" alt="A cool product">
  </div>
  <div class="product-info-container">
    <div class="product-title">Amazing Product Name</div>
    <div class="product-description">
      This is a short description of the product.
    </div>
    <div class="read-more-button" onclick="alert('Go to product page')">
      Read More
    </div>
  </div>
</div>

2. The `
` with a Life Vest

<!-- HTML with ARIA attributes -->
<div class="product-card" role="article" aria-labelledby="product-title-1">
  <div class="product-image-container">
    <img src="https://via.placeholder.com/300" alt="A cool product">
  </div>
  <div class="product-info-container">
    <div id="product-title-1" role="heading" aria-level="2">Amazing Product Name</div>
    <div class="product-description">
      This is a short description of the product.
    </div>
    <div 
      class="read-more-button" 
      role="link" 
      tabindex="0" 
      aria-label="Read more about Amazing Product Name"
      onclick="alert('Go to product page')"
    >
      Read More
    </div>
  </div>
</div>

3. The Best Practice (Semantic HTML)

<!-- HTML -->
<article class="product-card">
  <img src="https://via.placeholder.com/300" alt="A cool product">
  <div class="product-info-container">
    <h2>Amazing Product Name</h2>
    <p>
      This is a short description of the product.
    </p>
    <a href="#" class="read-more-button" aria-label="Read more about Amazing Product Name">
      Read More
    </a>
  </div>
</article>

Example 3: Modal Navigation

1. The `
` Nightmare

Note: This code requires custom JS to function.

<!-- HTML -->
<div class="modal-backdrop">
  <div class="modal-container">
    <div class="modal-header">
      <div class="modal-title">Settings</div>
      <div class="close-button" onclick="alert('Close Modal')">X</div>
    </div>
    <div class="modal-body-container">
      <div class="modal-nav">
        <div class="nav-item active" onclick="alert('Show General Tab')">General</div>
        <div class="nav-item" onclick="alert('Show Privacy Tab')">Privacy</div>
      </div>
      <div class="modal-content">
        <div class="content-tab">...General Content...</div>
        <div class="content-tab" style="display:none;">...Privacy Content...</div>
      </div>
    </div>
  </div>
</div>

2. The `
` with a Life Vest

<!-- HTML with ARIA attributes -->
<div 
  class="modal-backdrop" 
  role="dialog" 
  aria-modal="true" 
  aria-labelledby="modal-title"
>
  <div class="modal-container">
    <header class="modal-header">
      <h2 id="modal-title">Settings</h2>
      <button class="close-button" aria-label="Close modal" onclick="alert('Close Modal')">X</button>
    </header>
    <div class="modal-body-container">
      <nav class="modal-nav" role="tablist">
        <div role="tab" aria-selected="true">General</div>
        <div role="tab" aria-selected="false">Privacy</div>
      </nav>
      <div class="modal-content">
        <div role="tabpanel">...General Content...</div>
        <div role="tabpanel" style="display:none;">...Privacy Content...</div>
      </div>
    </div>
  </div>
</div>

3. The Best Practice (Semantic HTML)

<!-- HTML -->
<div class="modal-backdrop">
  <div class="modal-container" role="dialog" aria-modal="true" aria-labelledby="modal-title">
    <header class="modal-header">
      <h2 id="modal-title">Settings</h2>
      <button class="close-button" aria-label="Close modal" onclick="alert('Close Modal')">X</button>
    </header>
    <div class="modal-body-container">
      <nav class="modal-nav">
        <ul>
          <li><a href="#" class="nav-item">General</a></li>
          <li><a href="#" class="nav-item">Privacy</a></li>
        </ul>
      </nav>
      <div class="modal-content">
        <section><h3>General Settings</h3>...</section>
        <section style="display:none;"><h3>Privacy Settings</h3>...</section>
      </div>
    </div>
  </div>
</div>
